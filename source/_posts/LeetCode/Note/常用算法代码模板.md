# 深度优先
```code
var vistied Set
func dfs(node,vistied){
    vistied.add(node)
    //doSomething
    for next_node in node.chlidren() {
        if next_node  not in vistied {
                dfs(next_node,vistied)
        }
    }
}
```

# 广度优先

//广度搜索的模板
func BFS(graph,start,end){
	queue:=[]
	var visited  Set
	queue.append([start])
	visited.add(start)
	while !queue.isEmpty(){
		node=queue.pop()
		visited.add(node)
		doSomething(node)  //节点进行一些操作
		nodes:=generate_related_node(node) //去取出当前节点的后继节点，方法
