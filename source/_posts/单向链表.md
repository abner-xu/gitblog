package suanfa;


public class LinkList {
    //头节点
    private Node head;

    public LinkList() {
        head = null;
    }

    /**
     * 插入,再头节点进行插入
     */
    public void addHead(long value) {
        Node node = new Node(value);
        if (head == null) {
            head = node;
        } else {
            node.next = head;
            head = node;
        }

    }

    /**
     * 链表头删除
     *
     * @return
     */
    public Node delHead() {
        Node tmp = head.next;
        head = head.next;
        return tmp;
    }

    public void showAll() {
        Node current = head;
        while (current != null) {
            current.display();
            current = current.next;
        }
        System.out.println();
    }

    /**
     * 链表查找
     */
    public Node find(long value) {
        Node current = head;
        while (current != null && current.data != value) {
            if (current.next == null) {
                return null;
            }
            current = current.next;
        }
        return current;
    }

    public Node delValue(long value) {
        Node current = head;
        Node pre = head;//上一个节点
        while (current != null && current.data != value) {
            if (current.next == null) {
                return null;
            }
            pre = current;
            current = current.next;
        }

        if (current == head) {  //如果删除的是第一个节点
            head = current.next;
        } else {  //删除的不是第一个节点
            pre.next = current.next;
        }

        return current;

    }


}


/**
 * 相当于车厢
 */
class Node {
    //数据域
    public long data;
    //节点域
    public Node next;


    public Node(long value) {
        this.data = value;
    }

    /**
     * 显示
     */

    public void display() {
        System.out.print(data + " ");
    }
}
