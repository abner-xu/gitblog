package suanfa;

/**
 * 双向链表
 */
public class DoubleLinkList {
    private DoubleNode first;
    private DoubleNode last;

    public DoubleLinkList() {
        first = null;
        last = null;
    }

    /**
     * 头结点插入
     *
     * @param value
     */
    public void insertFirst(long value) {
        DoubleNode node = new DoubleNode(value);
        if (first==null) {
            last = node;
        }else{
            first.previous=node;
        }
        node.next = first;
        first = node;

    }

    /**
     * 尾节点插入
     *
     * @param value
     */
    public void insertLast(long value) {
        DoubleNode node = new DoubleNode(value);
        if (first==null) {
            first = node;
        } else {
            last.next = node;
            node.previous=last;
        }
        last=node;
    }

    /**
     * 返回删除的头部节点
     *
     * @return
     */
    public DoubleNode deleteFirst() {
        DoubleNode temp = first;
        if(first.next==null){//已经是最后一个节点了 
            last=null;
        }else{
            first.next.previous=null;
        }
        first=temp.next;
        return temp;
    }
    /**
     * 返回删除的尾部节点
     *
     * @return
     */
    public DoubleNode deleteLast() {
        DoubleNode temp = last;
        if(first.next==null){//已经是最后一个节点了 
            first=null;
        }else{
            last.previous.next=null;
        }
        last=last.previous;
        return temp;
    }

    

    public void showAll() {
        DoubleNode current = first;
        while (current != null) {
            current.display();
            current = current.next;
        }
        System.out.println();
    }

    /**
     * 链表查找
     */
    public DoubleNode find(long value) {
        DoubleNode current = first;
        while (current != null && current.data != value) {
            if (current.next == null) {
                return null;
            }
            current = current.next;
        }
        return current;
    }

    public DoubleNode delValue(long value) {
        DoubleNode current = first;
        while (current != null && current.data != value) {
            if (current.next == null) {
                return null;
            }
            current = current.next;
        }

        if (current == first) {  //如果删除的是第一个节点
            first = current.next;
        } else {  //删除的不是第一个节点
            current.previous.next = current.next;
        }

        return current;

    }

}







/**
 * 双向链表Node
 */
 public class DoubleNode {
    //数据域
    public long data;
    //下一节点域
    public DoubleNode next;
    //前一节点域
    public DoubleNode previous;


    public DoubleNode(long value) {
        this.data = value;
    }

    /**
     * 显示
     */

    public void display() {
        System.out.print(data + "\t");
    }
}
