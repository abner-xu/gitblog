```java
package stack;

public class MyStack {
    //底层实现是一个数组
    private long[] arr;
    //栈顶
    private int top;

    public MyStack(int max) {
        arr = new long[max];
        top = -1;
    }

    //添加
    public void push(int value) {
        arr[++top] = value;
    }

    //弹出
    public long pop() {
        return arr[top--];
    }
    //弹出
    public long peek() {
        return arr[top];
    }

    public boolean isEmpty() {
        return top == -1;
    }

    public boolean isFull() {
        return top == arr.length - 1;
    }
}

```
