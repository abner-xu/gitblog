package stack;

public class MyQueue {
    //底层实现是一个数组
    private long[] arr;
    //数据大小
    private int elements;
    //队头
    private int front;
    //队尾
    private int end;


    public MyQueue(int maxSize) {
        arr = new long[maxSize];
        elements = 0;
        front = 0;
        end = -1;
    }

    //插入数据，从队尾插入
    public void insert(long value) {
        //这里处理循环队列，删除的数据空位还可以继续使用,使用游标回归
        if(end==arr.length-1){
            end=-1;
        }
        arr[++end] = value;
        elements++;//计算有效数据
    }

    //删除数据，从队头
    public long remove() {
        long value=arr[front++];
        //这里处理循环队列，删除的数据空位还可以继续使用,使用游标回归
        if(front==arr.length){
            front=0;
        }
        elements--;
        return value;
    }

    //查看数据，从队头
    public long peek() {
        return arr[front];
    }

    public boolean isEmpty() {
        return elements == 0;
    }

    public boolean isFull() {
        return elements == arr.length;
    }

}
